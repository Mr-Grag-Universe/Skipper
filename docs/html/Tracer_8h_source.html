32int get_msb_ind(uint x) {
33 if (x == 0)
34 return -1; // Возвращает -1, если нет установленных битов
35
36 int msb_index = 0;
37 while (x >>= 1) {
38 msb_index++;
39 }
40 return msb_index;
41}

105 char* offset, uint32_t size, uint32_t ind, uint32_t reg_id) {
106 auto xax = DR_REG_XAX;
107 auto xbx = DR_REG_XBX;
108 auto xcx = DR_REG_XCX;
109 auto xdx = DR_REG_XDX;
110
111 // сохраняем регистры и флаги
112 dr_save_arith_flags(drcontext, bb, where, SPILL_SLOT_2); // по умолчанию кладёт в xax
113 dr_save_reg(drcontext, bb, where, xax, SPILL_SLOT_3);
114 dr_save_reg(drcontext, bb, where, xbx, SPILL_SLOT_4);
115 dr_save_reg(drcontext, bb, where, xcx, SPILL_SLOT_5);
116 dr_save_reg(drcontext, bb, where, xdx, SPILL_SLOT_6);
117
118 instr_t * instr;
119
120 // сохраняем значение исследуемого регистра в RCX
121 instr = XINST_CREATE_move(
122 drcontext,
123 opnd_create_reg(DR_REG_RCX),
124 opnd_create_reg(reg_id)
125 );
126 instrlist_meta_preinsert(bb, where, instr);
127
128 // =========================================================================
129 // get_msb_ind
130 // RCX - исследуемый регистр хранится
131 // RAX - возвращаемый результат
132 // создаём метки mgb_loop, msb_finish, msb_ret_minus_1
133
