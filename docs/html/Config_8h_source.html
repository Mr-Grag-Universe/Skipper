45 int get_opcode(std::string op_s) const {
46 if (op_s == "add" ) {
47 return OP_add;
48 } else if (op_s == "adc" ) {
49 return OP_adc;
50 } else if (op_s == "sub" ) {
51 return OP_sub;
52 } else if (op_s == "sbb" ) {
53 return OP_sbb;
54 } else if (op_s == "adcx") {
55 return OP_adcx;
56 } else if (op_s == "adox") {
57 return OP_adox;
58 } else if (op_s == "cmp" ) {
59 return OP_cmp;
60 } else if (op_s == "movq") {
61 return OP_movq;
62 } else {
63 main_logger.log_error("cannot translate {} to DR opcode - unknown!", op_s);
64 throw std::runtime_error("we do not know this opcode's name!");
65 return -1;
66 }
